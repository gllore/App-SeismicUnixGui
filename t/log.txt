   
user.pm
    Clear any pre-existing account information and files belonging to 'tester'
>getent passwd tester > /dev/null		
									if [ $? -eq 0 ]; then									
    								    sudo /usr/sbin/userdel -r tester   	
									else																
    									echo "No, the user does not exist"		
									fi
   Create temporary account for tester
>sudo /usr/sbin/luseradd --directory=/usr/local/pl/L_SU/t/tester --plainpassword=5 -g=tester --shell=/bin/bash tester
   
configuration.pm
    Create configuration subdirectories and files
>sudo mkdir -pv /usr/local/pl/L_SU/t/tester/.L_SU/configuration/active
    Create Servilleta_demos project in active directory
>sudo mkdir -pv /usr/local/pl/L_SU/t/tester/.L_SU/configuration/active/Servilleta_demos
    Create Servilleta_demos project in latent directory
>sudo mkdir -pv /usr/local/pl/L_SU/t/tester/.L_SU/configuration/Servilleta_demos
    Create demos project in latent directory
>sudo mkdir -pv /usr/local/pl/L_SU/t/tester/.L_SU/configuration/demos
    Write /usr/local/pl/L_SU/t/tester/.L_SU/configuration/active/Servilleta_demos/Project.config
>sudo cp /usr/local/pl/L_SU/t/Project.config4Servilleta_demos /usr/local/pl/L_SU/t/tester/.L_SU/configuration/active/Servilleta_demos/Project.config
    Write /usr/local/pl/L_SU/t/tester/.L_SU/configuration/Servilleta_demos/Project.config
>sudo cp /usr/local/pl/L_SU/t/Project.config4Servilleta_demos  /usr/local/pl/L_SU/t/tester/.L_SU/configuration/Servilleta_demos/Project.config
    Write /usr/local/pl/L_SU/t/tester/.L_SU/configuration/demos/Project.config
>sudo cp /usr/local/pl/L_SU/t/Project.config4demos /usr/local/pl/L_SU/t/tester/.L_SU/configuration/demos/Project.config
    Copy /usr/local/pl/L_SU/demo_projects/Servilleta_demos /usr/local/pl/L_SU/t/tester
>sudo cp -r /usr/local/pl/L_SU/demo_projects/Servilleta_demos /usr/local/pl/L_SU/t/tester
    Copy /usr/local/pl/L_SU/demo_projects/demos /usr/local/pl/L_SU/t/tester
>sudo cp -r /usr/local/pl/L_SU/demo_projects/demos /usr/local/pl/L_SU/t/tester
