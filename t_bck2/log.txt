--Clear any pre-existing account information and files belonging to 'tester'
>
	username='tester'
	getent passwd tester > /dev/null		
	if [ $? -eq 0 ]; then											
    	echo "A user = tester already exists. Can not continue"	
		exit 0														
																	
	else															
	    echo "A user= tester does not exist. OK to continue"		
    	sudo /usr/sbin/userdel -r tester   						
	fi
